# =============================================
# CONFIGURAÇÃO DNS COMPLETA - ULTRAZEND SMTP
# =============================================

# MX Records - Mail Exchange
ultrazend.com.br.           IN    MX    10    mail.ultrazend.com.br.
ultrazend.com.br.           IN    MX    20    mail2.ultrazend.com.br.
www.ultrazend.com.br.       IN    MX    10    mail.ultrazend.com.br.

# A Records - Servidores de Email
mail.ultrazend.com.br.      IN    A     [IP_SERVIDOR_PRINCIPAL]
mail2.ultrazend.com.br.     IN    A     [IP_SERVIDOR_BACKUP]

# DNS Reverso (PTR Records) - Para reputação
[IP_SERVIDOR_PRINCIPAL]     IN    PTR   mail.ultrazend.com.br.
[IP_SERVIDOR_BACKUP]        IN    PTR   mail2.ultrazend.com.br.

# SPF Record - Sender Policy Framework
ultrazend.com.br.           IN    TXT   "v=spf1 mx a ip4:[IP_SERVIDOR_PRINCIPAL] ip4:[IP_SERVIDOR_BACKUP] include:_spf.google.com ~all"

# DMARC Policy - Domain-based Message Authentication
_dmarc.ultrazend.com.br.    IN    TXT   "v=DMARC1; p=quarantine; rua=mailto:dmarc@ultrazend.com.br; ruf=mailto:dmarc@ultrazend.com.br; sp=quarantine; adkim=r; aspf=r; fo=1; rf=afrf; ri=86400"

# DKIM Selector - DomainKeys Identified Mail
default._domainkey.ultrazend.com.br.    IN    TXT   "v=DKIM1; k=rsa; t=s; p=[CHAVE_PUBLICA_DKIM_SERA_GERADA]"

# Subdominios adicionais para MX
smtp.ultrazend.com.br.      IN    A     [IP_SERVIDOR_PRINCIPAL]
submission.ultrazend.com.br. IN    A     [IP_SERVIDOR_PRINCIPAL]

# SRV Records para autodiscovery
_submission._tcp.ultrazend.com.br.  IN  SRV   0  1  587  mail.ultrazend.com.br.
_imap._tcp.ultrazend.com.br.        IN  SRV   0  1  143  mail.ultrazend.com.br.
_imaps._tcp.ultrazend.com.br.       IN  SRV   0  1  993  mail.ultrazend.com.br.

# TXT Record adicional para verificação
ultrazend.com.br.           IN    TXT   "ultrazend-verification=production-smtp-server-v1.0"

# INSTRUCOES DE CONFIGURACAO:
# 1. Substituir [IP_SERVIDOR_PRINCIPAL] pelo IP real do servidor
# 2. Substituir [IP_SERVIDOR_BACKUP] pelo IP do servidor secundário
# 3. A chave DKIM será gerada automaticamente pelo sistema
# 4. Configurar DNS reverso no provedor de hosting/VPS
# 5. Aguardar propagação DNS (24-48h para total propagação)
